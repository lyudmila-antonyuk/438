Задача №11

Напишите фрагмент кода (группу комманд), который будет присваивать значения 5 и 9 двум переменным, а затем третья переменная будет получать значение, равное сумме значений первых двух переменных. Вывести на экран значение третьей переменной.

let a, b;
a = +prompt("Введите первую переменную:");
if (!a && typeof a == "object") {
alert("Вы отменили!");
} else if (a == "") {
alert("Вы ничего не ввели!");
} else if (isNaN(a)) {
alert("Это не число!");
} else {
b = +prompt("Введите вторую переменную:");
if (!b && typeof b == "object") {
alert("Вы отменили!");
} else if (b == "") {
alert("Вы ничего не ввели!");
} else if (isNaN(b)) {
alert("Это не число!");
} else {
let sum = (a + b);
alert(`Сумма двух переменных: a = ${a}, b = ${b} равна ${sum}`);
}
}
